块设备关注 优化工作（合并 调序等），
字符设备直接调取mtd_info的read write就行。

MTD ( Memory Technology Device)存储技术设备

查看内核启动信息，从这里开始分析，

drivers\mtd\nand\s3c2410.c
"S3C24XX NAND Driver, (c) 2004 Simtec Electronics"


bus drv dev用的总线驱动设备模型，

drivers\mtd\nand\s3c2410.c
// 入口函数
static int __init s3c2410_nand_init(void)
{
	printk("S3C24XX NAND Driver, (c) 2004 Simtec Electronics\n");

	platform_driver_register(&s3c2412_nand_driver); // 注册平台驱动
	platform_driver_register(&s3c2440_nand_driver);
	return platform_driver_register(&s3c2410_nand_driver);
}

static struct platform_driver s3c2412_nand_driver = {
	.probe		= s3c2412_nand_probe,
	.remove		= s3c2410_nand_remove,
	.suspend	= s3c24xx_nand_suspend,
	.resume		= s3c24xx_nand_resume,
	.driver		= {
		.name	= "s3c2412-nand", // 会在probe中匹配name
		.owner	= THIS_MODULE,
	},
};

// probe函数的大致步骤
static int s3c24xx_nand_probe(struct platform_device *pdev,
			      enum s3c_cpu_type cpu_type)
{
	err = s3c2410_nand_inithw(info, pdev); // 初始化硬件
	s3c2410_nand_init_chip(info, nmtd, sets); // 初始化芯片
	// =============================================
	// 根据nand_chip的底层操作函数识别nand，构造=====mtd_info====
	nand_scan(&nmtd->mtd, (sets) ? sets->nr_chips : 1);	// drivers\mtd\nand\nand_base.c	
	// =============================================
		nand_scan_ident(mtd, maxchips);
			nand_set_defaults(chip, busw);
				// 片选
				if (!chip->select_chip) // 默认芯片选择没有做什么，不能用
					chip->select_chip = nand_select_chip;
				// 发命令发地址
				if (chip->cmdfunc == NULL)	
					chip->cmdfunc = nand_command; // 默认cmd会调用chip->cmd_ctrl
						chip->cmd_ctrl(mtd, command, ctrl);		
				// 读数据
				if (!chip->read_byte) 
					chip->read_byte = busw ? nand_read_byte16 : nand_read_byte;	
						nand_read_byte // 总线宽度是8 所以nand_read_byte
							readb(chip->IO_ADDR_R); // chip->IO_ADDR_R 数据寄存器的虚拟地址
				// 读数据
				if (!chip->read_buf)
					chip->read_buf = busw ? nand_read_buf16 : nand_read_buf;
						nand_write_buf // 总线宽度是8 所以nand_write_buf
							writeb(buf[i], chip->IO_ADDR_W); // 往这里写chip->IO_ADDR_W	
				// 判断状态
				if (chip->waitfunc == NULL)
					chip->waitfunc = nand_wait;		
						chip->dev_ready(mtd) // 需要这个函数			 				
			nand_get_flash_type(mtd, chip, busw, &nand_maf_id);	
				chip->select_chip(mtd, 0);	
				chip->cmdfunc(mtd, NAND_CMD_READID, 0x00, -1); // 读取id的命令			
				/* Read manufacturer and device IDs */
				*maf_id = chip->read_byte(mtd);	// 读取制造商信息id
				dev_id = chip->read_byte(mtd);  // 设备id
		nand_scan_tail(mtd);
			mtd->erase = nand_erase; // struct mtd_info *mtd
			mtd->read = nand_read;
			mtd->write = nand_write;
			nand_get_flash_type(mtd, chip, busw, &nand_maf_id);
				chip->select_chip(mtd, 0); // 选中芯片
				chip->cmdfunc(mtd, NAND_CMD_READID, 0x00, -1); // 发出命令 Send the command for reading device ID
				*maf_id = chip->read_byte(mtd); // 读取厂家id 
				dev_id = chip->read_byte(mtd); // 读取设备id  0xda 
				// Lookup the flash id 在数组里比较设备id
				for (i = 0; nand_flash_ids[i].name != NULL; i++) {
					if (dev_id == nand_flash_ids[i].id) {
						type =  &nand_flash_ids[i];
						break;
					}
				}				
	s3c2410_nand_add_partition(info, nmtd, sets);
		add_mtd_device(&mtd->mtd); // 添加分区添加设备 通知两个user 字符和块设备
			list_for_each(this, &mtd_notifiers) { // 这个链表在哪里设置呢mtd_notifiers，我们才知道add是干嘛的 drivers\mtd\mtdcore.c
						// 在register_mtd_user中添加，
						// register_mtd_user 在drivers\mtd\mtd_blkdevs.c 
						// drivers\mtd\mtdchar.c 都有调用。nand即可当作字符设备也可块设备 
						// 最终导致mtd_notify_add和blktrans_notify_add被调用
				struct mtd_notifier *not = list_entry(this, struct mtd_notifier, list);
				not->add(mtd);
				// mtd_notify_add和blktrans_notify_add
				// 先看字符设备 mtd_notify_add ，字符设备那一套
				class_device_create(mtd_class, NULL, MKDEV(MTD_CHAR_MAJOR, mtd->index*2),
				// 再看块设备 blktrans_notify_add
				list_for_each(this, &blktrans_majors) { // blktrans_majors这个链表在哪里设置呢,register_mtd_blktrans, register_mtd_blktrans被drivers\mtd\mtdblock.c和drivers\mtd\mtdblock_ro.c调用。 mtdblock_add_mtd 
					struct mtd_blktrans_ops *tr = list_entry(this, struct mtd_blktrans_ops, list);
					tr->add_mtd(tr, mtd);
					mtdblock_add_mtd (drivers\mtd\mtdblock.c)
						add_mtd_blktrans_dev // drivers\mtd\mtd_blkdevs.c
							alloc_disk
							set_capacity
							gd->queue = tr->blkcore_priv->rq;
							add_disk
	}
				
			    NULL, "mtd%d", mtd->index);
				
			}		
		
		
struct nand_flash_dev nand_flash_ids[] = {

#ifdef CONFIG_MTD_NAND_MUSEUM_IDS
	{"NAND 1MiB 5V 8-bit",		0x6e, 256, 1, 0x1000, 0},
	/* 2 Gigabit */
	{"NAND 256MiB 1,8V 8-bit",	0xAA, 0, 256, 0, LP_OPTIONS},
	{"NAND 256MiB 3,3V 8-bit",	0xDA, 0, 256, 0, LP_OPTIONS}, // nand 对应参数
