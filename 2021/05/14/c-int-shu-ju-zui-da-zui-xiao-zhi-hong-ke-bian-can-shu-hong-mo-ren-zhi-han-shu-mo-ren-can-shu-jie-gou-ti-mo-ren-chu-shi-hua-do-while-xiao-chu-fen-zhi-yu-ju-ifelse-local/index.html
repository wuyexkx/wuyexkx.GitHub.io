int型数据最大最小值宏表示

32位上

#include <stdio.h>

int main()
{
    #define NUM1 (~0U)      // 0U -> 0x00000000 -> ~ -> 0xFFFFFFFF(-1) -> >>1 -> 0x7FFFFFFF(INTMAX)
    #define NUM2 (~0U>>1)
    #define NUM3 (-NUM2-1)  //  -INTMAX-1

    printf("%%d:%20d  %20d  %20d\n", NUM1, NUM2, NUM3);
    printf("%%x:%20x  %20x  %20x\n", NUM1, NUM2, NUM3);
    printf("%%p:%20p  %20p  %20p\n", NUM1, NUM2, NUM3);

    return 0;
}

// %d:                  -1            2147483647           -2147483648
// %x:            ffffffff              7fffffff              80000000
// %p:    00000000FFFFFFFF      000000007FFFFFFF      0000000080000000

可变参数 , ## __VA_ARGS__

可变参数宏的引入就解决了这个问题：

  1. “…”只能放在参数宏形参列表的最后；
  2. 当用户的参数个数超过了规定的参数个数时，所有多出来的内容会一股脑的由__VA_ARGS__所背负；
  3. 当用户的参数个数正好等于形参的个数时，__VA_ARGS__就等效于一个空字符串。
  • 参数宏是通过“,”来作为分隔符来计算用户实际产传入了几个参数的.
  • 当你使用参数宏的时候，传入参数的个数（已“,”分开）必须与定义参数宏时候形参的数量完全一致；

当不一致的时候，预编译器可能不会报错，而是直接无视了你的参数宏——把它传递到编译的下一阶段，因而往往会被认作是一个函数——事实上这个函数是不存在的，因此在链接阶段会报告某某函数未定义的错误。

举例

#define safe_atom_code(...)
#define safe_atom_code(__CODE)

不同之处在于，前者括号里可以放包括","在内的几乎任意内容；而后者则完全不能容忍逗号的存在。

例子：

// 1 没有##试试
#define log_info(__STRING, ...)    printf(__STRING, __VA_ARGS__)
log_info("------------------------------------\r\n");
log_info(" Cycle Count : %d", total_cycle_cnt);
				| 展开后
				v
printf("------------------------------------\r\n",);			// 注意多个“,”
printf(" Cycle Count : %d", total_cycle_cnt);


// 2 加上##试试
#define log_info(__STRING, ...)    printf(__STRING,##__VA_ARGS__)
printf("------------------------------------\r\n");				// 清爽
printf(" Cycle Count : %d", total_cycle_cnt);


// __VA_ARGS__ 就是完全替换前面 ... 的，默认的
// 当然也可以为 ... 换个名字，
#define log_info(__STRING, args...)    printf(__STRING,##args)

默认参数

#define EXAMPLE(...)     ( 默认值 ,##__VA_ARGS__)
EXAMPLE();	被展开为 -->
 (默认值);
 
EXAMPLE(我们提供的值);  被展开为 -->
(默认值, 我们提供的值); // 根据逗号表达式的特性，此时，默认值会被丢弃掉
（有些编译器会报告表达式无效的warning，
这是正常的，因为编译器注意到“默认值”所代表的表达式实际上被丢弃了，它觉得我们写了一个无用的表达式）。

1. 举例 默认首个参数

#define FUNC(...) func((8, ##__VA_ARGS__))  // 注意内部还有个括号才能实现逗号表达式 

int func(int i)
{ printf("%d\n", i); }

// ===================== 调用 =========================
FUNC();		// 8
FUNC(88);	// 88

3 函数默认多个参数

// 运用三目运算符来选择参数，有则第一个，无则第二个
// #name[0] 先字符化 name，然后取出字符串的第0个元素，如果有则第一个参数，否则defval
// name + 0 如果name为空则三目表达式语法错误
#define DEFARG(name, defval) (#name[0] ? (name + 0) : (defval))
#define add(arg0, arg1, arg2) _add(DEFARG(arg0, 1), DEFARG(arg1, 2), DEFARG(arg2, 3))
void _add(int a, int b, int c)
{
    printf("%d  %d  %d\n", a, b, c);
}

// 调用
// func() 和 func( )参数解析都是 ""
add(,,);
add(2,,);
add(,3,);
add(,,4);
add(4,5,6);
    // 1  2  3
    // 2  2  3
    // 1  3  3
    // 1  2  4
    // 4  5  6

宏用 do { … } while (0)

// 为什么要在宏里用这个东西？
// #define AAA() printf("111\n"); printf("222\n");
// if (2 < 1) AAA();    // 下面两种情况被解析成 一部分满足条件
// if (2 < 1) 
//     AAA();
// if (2 < 1) {         // 我们认为会这样
//     AAA();
// }

// #define AAA() { printf("111\n"); printf("222\n"); }
// AAA();      // 通常调用时末尾加上";"，展开后 {}; // 现在的编译器会自动检测自动忽略分号，不会报错，但是我们还是希望能跑在老的编译器上。

// 下面两种写法可以避免多一个";"问题，也能解决可能出现的逻辑问题
// #define AAA()               \
//     do {                    \
//         printf("111\n");    \
//         printf("222\n");    \
//     } while (0)
#define AAA() do { printf("111\n"); printf("222\n"); } while (0)

1 if esle / switch case

逻辑简单，效率不高，分支判断过多。
时间长了，需求不断变更，程序不断迭代，这样会变得非常冗长而很难维护。

enum __player_cmd {
    CMD_PLAY,
    CMD_PAUSE,
    CMD_NEXT,
    CMD_PREV,
};

int16_t palyer_play(void* p)
{
    printf("palyer_play\n");
}
int16_t palyer_pause(void* p)
{
    printf("palyer_pause\n");
}
int16_t palyer_next(void* p)
{
    printf("palyer_next\n");
}
int16_t palyer_prev(void* p)
{
    printf("palyer_prev\n");
}

int main()
{
    base_data b_d;
    b_d.data = 0x12345678;
    int16_t cmd = CMD_NEXT;

// 常规做法 1   if else 
    if (cmd == CMD_PLAY) {
        palyer_play(&b_d);
    } else if (cmd == CMD_PAUSE) {
        palyer_pause(&b_d);
    } else if (cmd == CMD_NEXT) {
        palyer_next(&b_d);
    } else if (cmd == CMD_PREV) {
        palyer_prev(&b_d);
    }

// 常规做法 2   switch
    cmd = CMD_PLAY;
    switch (cmd) {
    case CMD_PLAY:
        palyer_play(&b_d);
        break;
    case CMD_PAUSE:
        palyer_pause(&b_d);
        break;
    case CMD_NEXT:
        palyer_next(&b_d);
        break;
    case CMD_PREV:
        palyer_prev(&b_d);
        break;                    
    default:
        break;
    }

    return 0;
}

2 函数指针，for循环

将cmd和对应func存入数组，遍历数组匹配cmd执行函数。
需要遍历，还要考虑enum中变量与函数的对应关系，效率跟常规做法差不多，只是维护方便。

typedef int16_t(*pfunc)(void* p);
typedef struct __cmd_func {
    int16_t cmd;
    pfunc func;
} cmd_func; 
cmd_func cmd_funcs[] = {
    {CMD_PLAY, player_play},
    {CMD_PAUSE, player_pause},
    {CMD_NEXT, player_next},
    {CMD_PREV, player_prev},
};

int main()
{
// 函数指针做法 1，需要遍历，效率跟常规做法差不多，只是维护方便
    #define ARR_LEN(array) (sizeof(array) / sizeof(*array))
    for (int i = 0; i < ARR_LEN(cmd_funcs); i++) {
        if (cmd == cmd_funcs[i].cmd && cmd_funcs[i].func != NULL) {
            cmd_funcs[i].func(&b_d);
            break;
        }
    }
    
	return 0;
}

3 用macro生成enum和函数指针array

利用宏分别展开cmd 和 func 到 enum 和 array，
无需遍历，下标索引，效率高，

3.1 简述原理，展开宏

// 写完 cmd 和 func 只需要在这里绑定就 ok
#define CMD_FUNC \
    DEF_X(PLAY, player_play)    \
    DEF_X(PAUSE, player_pause)  \
    DEF_X(NEXT, player_next)    \
    DEF_X(PREV, player_prev)    \

enum cmd {
#define DEF_X(a, b) a,
    CMD_FUNC
    CMD_MAX,
#undef DEF_X
};

const pfunc funcs[] = {
#define DEF_X(a, b) b,
    CMD_FUNC
#undef DEF_X
}; 
/* 自动展开之后是下面这样
enum cmd {

    PLAY, PAUSE, NEXT, PREV,
    CMD_MAX,

};

const pfunc funcs[] = {

    player_play, player_pause, player_next, player_prev,

};
*/
void cmd_handle(int16_t cmd, void* p)
{
    if (cmd < CMD_MAX) {
        funcs[cmd](p);
    } else {
        printf("cmd %d invalid!\n", cmd);
    }
}

int main()
{
	cmd_handle(cmd, &b_d);
    return 0;
}

3.2 一个小例子，根据CMD执行对应函数


#include <stdio.h>
#include <unistd.h>

// 需要执行的函数1 2 3
int mmd_func1(void* p)
{
    (*((int*)p))++;
	printf("%d mmd_func1\n", *((int*)p));
    return 1;
}
int mmd_func2(void* p)
{
    (*((int*)p))++;
	printf("%d mmd_func2\n", *((int*)p));
    return 2;
}
int mmd_func3(void* p)
{
    (*((int*)p))++;
	printf("%d mmd_func3\n", *((int*)p));
    return 3;
}

// 通过DEF_X宏分别提取arg1, arg2
#define CMD_FUNCS \
	DEF_X(MMD1, mmd_func1)	\
	DEF_X(MMD2, mmd_func2)	\
	DEF_X(MMD3, mmd_func3)	\
// 通过定义不同的宏DEF_X选择提取DEF_X的arg1，并成为enum的项，默认0～N_CMD_MAX
typedef enum _cmd {
#define DEF_X(a, b) a,	
	CMD_FUNCS
	N_CMD_MAX,
#undef DEF_X
} cmd;
// 定义了一个函数指针数组，数组元素为指向int(*)(void*)的指针
// 通过定义不同的宏DEF_X选择提取DEF_X的arg2，并成为数组的元素，元素为函数指针
int(*arr_funcs[])(void*) = {
#define DEF_X(a, b) b,
	CMD_FUNCS
#undef DEF_X
};

// 根据cmd选择函数执行
void exec_funcs(cmd n, void* p)
{
	if (n < N_CMD_MAX && n >= 0)
		arr_funcs[n](p);
    else
        printf("cmd %d invalid!\n", n);
}


int main()
{
	int data = 0;
	int* p_data = &data;

	while (1) {
        // 一次执行函数1 2 ... N_CMD_MAX
		for (int i = 0; i < N_CMD_MAX; i++) {
			exec_funcs(i, p_data);
            sleep(1);
        }
        
	}	

	return 0;
}
